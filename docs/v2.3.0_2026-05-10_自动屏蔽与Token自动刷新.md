**自动过滤Hook** (`hooks/useAutoFilter.ts`)
```typescript
export function useAutoFilter(articles: Article[]) {
  const { autoHideRead, autoHideBlocked } = useSettingsStore();
  const { readHistory } = useArticleStore();

  // 已读文章ID集合
  const readArticleIds = useMemo(() => {
    const ids = new Set<number>();
    readHistory.forEach(article => ids.add(article.id));
    return ids;
  }, [readHistory]);

  // 过滤后的文章列表
  const filteredArticles = useMemo(() => {
    if (!autoHideRead && !autoHideBlocked) return articles;
    
    // 后端已过滤不看文章，前端只需过滤已读
    if (autoHideRead) {
      return articles.filter(article => !readArticleIds.has(article.id));
    }
    
    return articles;
  }, [articles, autoHideRead, autoHideBlocked, readArticleIds]);

  return { filteredArticles, autoHideRead, autoHideBlocked };
}
```

**自动补充机制** (`app/page.tsx`)
```typescript

```


### 2. 忽略文章页面 📋

#### 功能描述
在"我的"页面新增"忽略文章"入口，显示所有用户标记为"不看"的文章列表，支持恢复显示。

#### 实现细节

**页面路由** (`app/me/blocked/page.tsx`)
```typescript
export default function BlockedPage() {
  const [blockedArticles, setBlockedArticles] = useState<BlockedArticle[]>([]);
  
  // 获取忽略文章列表
  const fetchBlockedArticles = useCallback(async () => {
    const response = await api.get("/user/blocks?page=1&size=100");
    setBlockedArticles(response.items || []);
  }, []);

  // 恢复显示（取消不看）
  const handleUnblock = async (articleId: number) => {
    const response = await api.post(`/user/blocks/${articleId}`);
    if (response.success && !response.is_blocked) {
      setBlockedArticles(prev => prev.filter(item => item.article_id !== articleId));
    }
  };

  return (
    <div className="flex flex-col min-h-screen pb-14">
      <header>
        <h1>忽略文章</h1>
        <span>{blockedArticles.length} 篇</span>
      </header>
      
      {/* 文章列表 */}
      {blockedArticles.map((item) => (
        <div key={item.interaction_id}>
          <div>
            <h3>{item.article?.title}</h3>
            <p>{item.article?.source}</p>
          </div>
          <button onClick={() => handleUnblock(item.article_id)}>
            恢复显示
          </button>
        </div>
      ))}
    </div>
  );
}
```

## 🔧 功能优化

### 3. Token自动刷新机制 🔐

#### 问题描述
- Access Token有效期仅30分钟，用户需要频繁登录
- Token过期后直接返回401，体验差
- 没有自动续期机制

#### 解决方案

实现三层防护机制：

**第一层：API层自动刷新** (`lib/api.ts`)
```typescript
// 并发锁，防止多个请求同时刷新
let refreshPromise: Promise<boolean> | null = null;

async function refreshAccessToken(): Promise<boolean> {
  if (refreshPromise) return refreshPromise;
  
  refreshPromise = (async () => {
    const refreshToken = getRefreshToken();
    const response = await fetch('/api/v1/auth/refresh', {
      method: 'POST',
      body: JSON.stringify({ refresh_token: refreshToken })
    });
    
    if (response.ok) {
      const data = await response.json();
      setTokens(data.access_token, data.refresh_token);
      return true;
    }
    return false;
  })();
  
  return refreshPromise;
}

// 401自动处理
async function apiFetch<T>(path: string, options?: RequestInit) {
  const res = await fetch(url, { ...options, headers });
  
  if (res.status === 401 && token && !options?.skipAuthRefresh) {
    const refreshed = await refreshAccessToken();
    
    if (refreshed) {
      // 刷新成功，使用新token重试
      const newToken = getToken();
      headers["Authorization"] = `Bearer ${newToken}`;
      const retryRes = await fetch(url, { ...options, headers });
      return retryRes.json();
    } else {
      // 刷新失败，清除登录状态
      localStorage.removeItem('rss-auth-storage');
      throw new ApiError(401, "登录已过期，请重新登录");
    }
  }
  
  return res.json();
}
```

**第二层：Token管理器** (`hooks/useTokenManager.ts`)
```typescript
export function useTokenManager() {
  const { isAuthenticated, checkAuth } = useAuthStore();

  useEffect(() => {
    if (!isAuthenticated) return;

    // 每20分钟检查一次token
    const checkAndRefreshToken = async () => {
      const isValid = await checkAuth();
      if (!isValid) {
        console.log("Token will be refreshed on next API call");
      }
    };

    // 初始检查 + 定期检查
    checkAndRefreshToken();
    const interval = setInterval(checkAndRefreshToken, 20 * 60 * 1000);

    return () => clearInterval(interval);
  }, [isAuthenticated, checkAuth]);

  // 页面可见性变化时检查
  useEffect(() => {
    const handleVisibilityChange = () => {
      if (document.visibilityState === "visible") {
        checkAuth();
      }
    };

    document.addEventListener("visibilitychange", handleVisibilityChange);
    return () => document.removeEventListener("visibilitychange", handleVisibilityChange);
  }, [isAuthenticated, checkAuth]);
}
```

**第三层：增强checkAuth** (`store/useAuthStore.ts`)
```typescript
checkAuth: async () => {
  const { token, refreshToken } = get();
  
  // 验证token
  const res = await fetch('/api/v1/auth/verify', {
    headers: { Authorization: `Bearer ${token}` }
  });
  const data = await res.json();
  
  if (data.is_valid) return true;
  
  // Token无效，尝试刷新
  if (refreshToken) {
    const refreshRes = await fetch('/api/v1/auth/refresh', {
      method: 'POST',
      body: JSON.stringify({ refresh_token: refreshToken })
    });
    
    if (refreshRes.ok) {
      const refreshData = await refreshRes.json();
      if (refreshData.access_token) {
        await get().login(refreshData.access_token, refreshData.refresh_token);
        return true;
      }
    }
  }
  
  // 刷新失败，登出
  get().logout();
  return false;
}
```

#### Token生命周期

```
登录成功
  ↓
access_token (30分钟) + refresh_token (7天)
  ↓
正常使用（每次请求自动携带access_token）
  ↓
20分钟后 → 主动检查token状态
  ↓
30分钟后 → access_token过期
  ├─ API返回401
  ├─ 自动调用 /auth/refresh
  ├─ 获取新的 access_token
  └─ 重试原请求（用户无感知）✅
  ↓
7天后 → refresh_token也过期
  └─ 需要重新登录
```

#### 用户体验提升

| 场景 | 优化前 | 优化后 |
|------|--------|--------|
| **正常使用** | 30分钟后掉线 ❌ | 7天内免登录 ✅ |
| **token过期** | 需要手动登录 ❌ | 自动刷新，无感知 ✅ |
| **并发请求** | 多次刷新token ❌ | 并发锁，只刷新一次 ✅ |
| **网络波动** | 可能掉线 ❌ | 容错处理，保持登录 ✅ |
| **切回应用** | token可能已过期 ❌ | 自动检查并刷新 ✅ |

### 4. 不看按钮优化 🎯

#### 问题描述
点击"不看"按钮后强制返回首页，打断用户阅读流程。

#### 优化方案

**优化前**
```typescript
const handleBlockToggle = async () => {
  const response = await api.post(`/user/blocks/${article.id}`);
  if (response.is_blocked) {
    showToastMessage("已加入不看列表");
    // 3秒后返回首页 ❌
    setTimeout(() => router.push("/"), 3000);
  }
};
```

**优化后**
```typescript
const handleBlockToggle = async () => {
  const response = await api.post(`/user/blocks/${article.id}`);
  if (response.is_blocked) {
    showToastMessage("已加入不看列表");
    
    // 自动跳转到下一篇文章 ✅
    const nextId = getNextId(article.id);
    if (nextId) {
      // 有下一篇文章，1秒后跳转
      setTimeout(() => {
        router.push("/article/" + nextId);
      }, 1000);
    } else {
      // 没有下一篇，返回列表页
      // 列表页会自动过滤不看文章并补充
      setTimeout(() => {
        router.back();
      }, 1000);
    }
  }
};
```

#### 工作流程

**场景1：有下一篇文章**
```
文章详情页（当前：文章5/10）
    ↓
用户点击"不看"
    ↓
Toast提示："已加入不看列表"
    ↓
1秒后自动跳转
    ↓
文章详情页（下一篇：文章6/10）✅
```

**场景2：最后一篇文章**
```
文章详情页（当前：文章10/10）
    ↓
用户点击"不看"
    ↓
Toast提示："已加入不看列表"
    ↓
1秒后返回列表页
    ↓
列表页自动过滤不看文章
    ↓
如果文章不足自动补充 ✅
```
